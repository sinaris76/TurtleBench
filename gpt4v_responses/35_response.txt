```python
import turtle

# Set up the screen and turtle
screen = turtle.Screen()
screen.bgcolor("white")
t = turtle.Turtle()

# Set the speed and hide the turtle
t.speed(0)
t.hideturtle()

# Draw the shape
t.up()
t.goto(0, -100)  # Start at the base of the triangle, adjust as needed
t.down()

# Draw triangle
t.seth(0)
t.forward(100)
t.left(120)
t.forward(200)
t.left(120)
t.forward(200)
t.left(120)
t.forward(100)

# Draw semi-circle
t.seth(90)
t.circle(100, 180)

# Finish drawing
turtle.done()
```
