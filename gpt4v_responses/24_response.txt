```python
import turtle

# Setup the screen
wn = turtle.Screen()

# Create a turtle instance
t = turtle.Turtle()

# Set the speed and hide the turtle
t.speed(0)
t.hideturtle()

# Draw the triangles
def draw_triangle(base_length):
    for _ in range(3):
        t.forward(base_length)
        t.left(120)

# Draw the larger triangle
draw_triangle(200)

# Position for the smaller triangle
t.forward(200)
t.left(120)
t.forward(100)
t.left(120)

# Draw the smaller triangle
draw_triangle(100)

# Finish drawing
turtle.done()
```
