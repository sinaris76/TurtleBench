```python
import turtle

# Set up the screen
wn = turtle.Screen()
wn.bgcolor("white")

# Create a turtle
t = turtle.Turtle()
t.speed(0)  # Highest speed
t.hideturtle()

# Drawing the square
for _ in range(4):
    t.forward(100)
    t.left(90)

# Drawing the triangles
t.goto(0, 100)
t.goto(100, 0)
t.goto(0, 0)
t.goto(100, 100)

# Finish drawing
turtle.done()
```
