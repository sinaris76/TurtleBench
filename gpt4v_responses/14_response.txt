```python
import turtle

# Set up the screen
wn = turtle.Screen()
wn.bgcolor("white")

# Create a turtle
t = turtle.Turtle()
t.speed(0)  # Fastest speed
t.hideturtle()
t.pensize(2)

# Draw the figure
for _ in range(3):
    t.forward(100)
    t.left(90)
t.forward(100)
t.right(45)
t.forward(70.7107)  # Length of the diagonal of the square
t.right(90)
t.forward(70.7107)
t.right(45)
t.forward(100)
for _ in range(2):
    t.left(90)
    t.forward(100)

# Finish drawing
wn.mainloop()
```
