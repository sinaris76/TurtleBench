```python
import turtle

# Set up the screen
wn = turtle.Screen()
wn.bgcolor("white")

# Create a turtle
shape_turtle = turtle.Turtle()
shape_turtle.speed(0)  # Highest speed
shape_turtle.hideturtle()  # Hide turtle

# Draw a square
for _ in range(4):
    shape_turtle.forward(100)
    shape_turtle.left(90)

# Draw a triangle
shape_turtle.forward(100)
shape_turtle.left(135)
shape_turtle.forward(70.71)
shape_turtle.left(90)
shape_turtle.forward(70.71)
shape_turtle.left(135)

# Finish drawing
shape_turtle.done()
```
