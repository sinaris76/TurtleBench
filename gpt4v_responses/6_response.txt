```python
import turtle

# Set up the screen
wn = turtle.Screen()
wn.bgcolor("white")

# Create a turtle
t = turtle.Turtle()
t.speed(0)
t.hideturtle()

# Draw two triangles
for _ in range(2):
    t.forward(100) # base side length of the triangle
    t.left(120)
    t.forward(100)
    t.left(120)
    t.forward(100)
    t.left(60)

# Close the window on a click
wn.exitonclick()
```
