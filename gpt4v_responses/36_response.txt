```python
import turtle

screen = turtle.Screen()
screen.tracer(0, 0)  # Disable animation

t = turtle.Turtle()
t.speed(0)  # Set the speed to the fastest
t.hideturtle()  # Hide the turtle cursor

# Draw the first circle
t.penup()
t.goto(0, -200)
t.pendown()
t.circle(200)

# Draw the second circle
t.penup()
t.goto(250, -70)
t.pendown()
t.circle(70)

screen.update()  # Update the screen with the drawn shapes
turtle.done()
```
