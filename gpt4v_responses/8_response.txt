```python
import turtle

turtle.speed('fastest')  # Set the speed to the fastest
turtle.hideturtle()  # Hide the turtle

# Draw the square in the middle
for _ in range(4):
    turtle.forward(100)
    turtle.left(90)

# Draw the triangles
for _ in range(4):
    turtle.forward(100)
    turtle.left(45)
    turtle.forward(70.71)  # (100 / sqrt(2)) to reach the corner of the square
    turtle.left(135)
    turtle.forward(100)
    turtle.right(90)
    turtle.forward(70.71)  # Same length to go back to the square
    turtle.right(45)

turtle.done()
```
