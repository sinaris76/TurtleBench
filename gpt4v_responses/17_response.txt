```python
import turtle

screen = turtle.Screen()
screen.setup(800, 600)

t = turtle.Turtle()
t.speed(0)
t.hideturtle()

# Draw the main face circle
t.penup()
t.goto(0, -100)
t.pendown()
t.circle(100)

# Draw the left eye
t.penup()
t.goto(-35, 20)
t.pendown()
t.circle(10)

# Draw the right eye
t.penup()
t.goto(35, 20)
t.pendown()
t.circle(10)

# Draw the nose which is a triangle
t.penup()
t.goto(-15, -20)
t.pendown()
t.begin_fill()
t.forward(30)
t.left(120)
t.forward(30)
t.left(120)
t.forward(30)
t.end_fill()

turtle.done()
```
