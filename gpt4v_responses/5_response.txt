```python
import turtle

def draw_triangle(base_length):
    for _ in range(3):
        turtle.forward(base_length)
        turtle.left(120)

turtle.speed(0)
turtle.hideturtle()

base_length = 100  # Define the base_length variable for the size of the triangle

draw_triangle(base_length)
turtle.right(60)
turtle.forward(base_length)
turtle.right(120)
draw_triangle(base_length)
turtle.forward(base_length / 2)
turtle.left(60)
draw_triangle(base_length / 2)

turtle.done()
```
