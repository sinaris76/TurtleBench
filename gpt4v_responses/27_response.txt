```python
import turtle

screen = turtle.Screen()
screen.setup(width=700, height=700)

t = turtle.Turtle()
t.speed("fastest")
t.hideturtle()

# Draw square
t.penup()
t.goto(-50, -50)
t.pendown()
for _ in range(4):
    t.forward(100)
    t.left(90)

# Draw circle
t.penup()
t.goto(0, -50)
t.pendown()
t.circle(50)

screen.mainloop()
```
