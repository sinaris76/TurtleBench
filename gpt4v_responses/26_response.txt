```python
import turtle

screen = turtle.Screen()
t = turtle.Turtle()
t.speed(0)
t.hideturtle()

def draw_rectangle():
    for _ in range(2):
        t.forward(100)
        t.left(90)
        t.forward(50)
        t.left(90)

t.penup()
t.goto(-150, 25) # starting point of the first rectangle
t.pendown()

for _ in range(3):
    draw_rectangle()
    t.forward(100)

screen.mainloop()
```
