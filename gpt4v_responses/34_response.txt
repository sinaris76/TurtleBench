```python
import turtle

# Set up the screen
wn = turtle.Screen()
wn.bgcolor("white")

# Create a turtle
shape_turtle = turtle.Turtle()
shape_turtle.speed('fastest')  # Set the speed to the highest
shape_turtle.hideturtle()  # Hide the turtle

# Draw the shape
for _ in range(4):
    shape_turtle.forward(100)
    shape_turtle.right(90)
    shape_turtle.forward(25)
    shape_turtle.right(90)
    shape_turtle.forward(75)
    shape_turtle.left(90)
    shape_turtle.forward(25)
    shape_turtle.left(90)
    shape_turtle.forward(25)
    shape_turtle.right(90)

    # Move to the initial position for the next arm
    shape_turtle.left(90)
    shape_turtle.forward(75)
    shape_turtle.left(90)

# Finish drawing
turtle.done()
```
