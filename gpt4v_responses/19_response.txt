```python
import turtle

# function to draw a hexagon
def draw_hexagon():
    for _ in range(6):
        turtle.forward(100)
        turtle.left(60)

# setup turtle environment
turtle.speed(0)
turtle.hideturtle()

# draw top hexagon
turtle.penup()
turtle.goto(0, 100)
turtle.pendown()
turtle.left(30)
draw_hexagon()

# draw middle left hexagon
turtle.penup()
turtle.goto(-50, -50)
turtle.setheading(0)
turtle.pendown()
draw_hexagon()

# draw middle right hexagon
turtle.penup()
turtle.goto(50, -50)
turtle.setheading(0)
turtle.pendown()
draw_hexagon()

# draw bottom left hexagon
turtle.penup()
turtle.goto(-100, -200)
turtle.setheading(0)
turtle.pendown()
draw_hexagon()

# draw bottom right hexagon
turtle.penup()
turtle.goto(100, -200)
turtle.setheading(0)
turtle.pendown()
draw_hexagon()

# end the turtle drawing and keep the window open
turtle.done()
```
