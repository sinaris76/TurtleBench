import turtle

def draw_circle_with_diagonals(radius):
    turtle.penup()
    turtle.goto(0, -radius)
    turtle.pendown()
    turtle.circle(radius)
    for _ in range(4):
        turtle.penup()
        turtle.goto(0, 0)
        turtle.pendown()
        turtle.forward(radius)
        turtle.penup()
        turtle.backward(radius)
        turtle.left(90)

turtle.speed('fastest')
turtle.hideturtle()

draw_circle_with_diagonals(100)
turtle.penup()
turtle.goto(0, 200)
turtle.pendown()
draw_circle_with_diagonals(100)

turtle.done()
