```python
import turtle

screen = turtle.Screen()
t = turtle.Turtle()
t.speed(0)
t.hideturtle()

# Draw circle
t.circle(100)

# Draw the lines
for angle in [0, 45, 90, 135, 180, 225, 270, 315]:
    t.setheading(angle)
    t.penup()
    t.goto(0, 0)
    t.pendown()
    t.forward(100)

screen.mainloop()
```
