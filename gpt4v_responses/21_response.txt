```python
import turtle

def draw_branch(t, branch_length, shorten_by, angle):
    if branch_length > 4:
        t.forward(branch_length)
        new_length = branch_length - shorten_by

        t.left(angle)
        draw_branch(t, new_length, shorten_by, angle)

        t.right(angle * 2)
        draw_branch(t, new_length, shorten_by, angle)

        t.left(angle)
        t.backward(branch_length)

turtle.setup(800, 600)
screen = turtle.Screen()
t = turtle.Turtle()
t.speed('fastest')
turtle.hideturtle()

t.left(90)
draw_branch(t, 100, 15, 30)

screen.exitonclick()
```
