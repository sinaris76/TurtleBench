```python
import turtle

# create turtle object
t = turtle.Turtle()

# set the speed and hide the turtle
t.speed(0)
t.hideturtle()

# draw a square
for _ in range(4):
    t.forward(100)
    t.left(90)

# draw the diagonals
t.goto(0, 100)
t.goto(100, 0)
t.goto(0, 0)
t.goto(100, 100)

# finish drawing
turtle.done()
```
