```python
import turtle

def draw_triangle(t, length):
    for _ in range(3):
        t.forward(length)
        t.left(120)

screen = turtle.Screen()
t = turtle.Turtle()
t.speed(0)
t.hideturtle()

side_length = 100  # Adjust the length as needed

# Draw four triangles in a row
for _ in range(4):
    draw_triangle(t, side_length)
    t.forward(side_length)

screen.mainloop()
```
