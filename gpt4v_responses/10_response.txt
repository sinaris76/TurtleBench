```python
import turtle

screen = turtle.Screen()
screen.tracer(0, 0)

t = turtle.Turtle()
t.speed(0)
t.hideturtle()

# drawing the square
t.penup()
t.goto(-100, 100)
t.pendown()
for _ in range(4):
    t.forward(200)
    t.right(90)

# drawing the circle
t.penup()
t.goto(0, -100)
t.setheading(0)
t.pendown()
t.circle(100)

screen.update()
turtle.done()
```
