```python
import turtle

# Set up the screen
wn = turtle.Screen()
wn.bgcolor("white")

# Create a turtle named "t"
t = turtle.Turtle()
t.speed(0)  # Set the speed to the fastest
t.hideturtle()  # Hide the turtle

# Draw a square and a diagonal
for _ in range(4):
    t.forward(100)
    t.left(90)

t.goto(0, 0)  # Go back to the starting point
t.goto(100, 100)  # Draw the diagonal

# Finish drawing
wn.mainloop()
```
