```python
import turtle

t = turtle.Turtle()
t.speed("fastest")
t.hideturtle()

# Draw the first square
for _ in range(4):
    t.forward(100)
    t.left(90)

# Move to the position to draw the second square
t.forward(100)
t.right(90)
t.forward(100)
t.left(90)

# Draw the second square
for _ in range(4):
    t.forward(100)
    t.left(90)

turtle.done()
```
