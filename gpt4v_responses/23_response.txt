```python
import turtle

turtle.speed(0)
turtle.hideturtle()

for _ in range(4):
    turtle.forward(100)
    turtle.right(90)
    turtle.forward(25)
    turtle.right(90)
    turtle.forward(25)
    turtle.left(90)
    turtle.forward(25)
    turtle.left(90)
    turtle.forward(25)
    turtle.right(90)

turtle.done()
```
