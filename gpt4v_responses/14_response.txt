```python
import turtle

def draw_square(t, size):
    for _ in range(4):
        t.forward(size)
        t.left(90)

def draw_rotated_square(t, size, angle):
    t.left(angle)
    draw_square(t, size)
    t.right(angle)

t = turtle.Turtle()
t.speed('fastest')
t.hideturtle()

draw_square(t, 100)          # Draw first square
t.forward(100)               # Move to position for the second square
draw_rotated_square(t, 100, 45) # Draw second, rotated square

turtle.done()
```
